;Contrato: tira-empty: lista-URL-acessos um-número-de-acessos -> lista-URL-acessos
;Objetivo: adiciona um-número-de-acessos ao final da lista de entrada
;Cabeçalho e Corpo:

(define (tira-empty lista-URL-acessos um-número-de-acessos)
  (cond
    [(empty? lista-URL-acessos) (cons um-número-de-acessos empty)]
    [else (cons (first lista-URL-acessos) (tira-empty (rest lista-URL-acessos) um-número-de-acessos))]))
